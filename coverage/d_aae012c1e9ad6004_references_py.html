106def relative_url(url_a: str, url_b: str) -> str: 

107 """Compute the relative path from URL A to URL B. 

108 

109 Arguments: 

110 url_a: URL A. 

111 url_b: URL B. 

112 

113 Returns: 

114 The relative URL to go from A to B. 

115 """ 

116 parts_a = url_a.split("/") 

117 url_b, anchor = url_b.split("#", 1) 

118 parts_b = url_b.split("/") 

119 

120 # remove common left parts 

121 while parts_a and parts_b and parts_a[0] == parts_b[0]: 

122 parts_a.pop(0) 

123 parts_b.pop(0) 

124 

125 # go up as many times as remaining a parts' depth 

126 levels = len(parts_a) - 1 

127 parts_relative = [".."] * levels + parts_b # noqa: WPS435 

128 relative = "/".join(parts_relative) 

129 return f"{relative}#{anchor}"

132def fix_ref(url_mapper: Callable[[str], str], unmapped: List[str]) -> Callable: # noqa: WPS212,WPS231 

133 """Return a `repl` function for [`re.sub`](https://docs.python.org/3/library/re.html#re.sub). 

134 

135 In our context, we match Markdown references and replace them with HTML links. 

136 

137 When the matched reference's identifier was not mapped to an URL, we append the identifier to the outer 

138 `unmapped` list. It generally means the user is trying to cross-reference an object that was not collected 

139 and rendered, making it impossible to link to it. We catch this exception in the caller to issue a warning. 

140 

141 Arguments: 

142 url_mapper: A callable that gets an object's site URL by its identifier, 

143 such as [mkdocs_autorefs.plugin.AutorefsPlugin.get_item_url][]. 

144 unmapped: A list to store unmapped identifiers. 

145 

146 Returns: 

147 The actual function accepting a [`Match` object](https://docs.python.org/3/library/re.html#match-objects) 

148 and returning the replacement strings. 

149 """ 

150 

151 def inner(match: Match): # noqa: WPS212,WPS430 

152 identifier = match["identifier"] 

153 title = match["title"] 

154 kind = match["kind"] 

155 

156 try: 

157 url = url_mapper(unescape(identifier)) 

158 except KeyError: 

159 if kind == "autorefs-optional": 

160 return title 

161 elif kind == "autorefs-optional-hover": 

162 return f'<span title="{identifier}">{title}</span>' 

163 unmapped.append(identifier) 

164 if title == identifier: 

165 return f"[{identifier}][]" 

166 return f"[{title}][{identifier}]" 

167 

168 parsed = urlsplit(url) 

169 external = parsed.scheme or parsed.netloc 

170 classes = ["autorefs", "autorefs-external" if external else "autorefs-internal"] 

171 class_attr = " ".join(classes) 

172 if kind == "autorefs-optional-hover": 

173 return f'<a class="{class_attr}" title="{identifier}" href="{escape(url)}">{title}</a>' 

174 return f'<a class="{class_attr}" href="{escape(url)}">{title}</a>' 

175 

176 return inner

179def fix_refs(html: str, url_mapper: Callable[[str], str]) -> Tuple[str, List[str]]: 

180 """Fix all references in the given HTML text. 

181 

182 Arguments: 

183 html: The text to fix. 

184 url_mapper: A callable that gets an object's site URL by its identifier, 

185 such as [mkdocs_autorefs.plugin.AutorefsPlugin.get_item_url][]. 

186 

187 Returns: 

188 The fixed HTML. 

189 """ 

190 unmapped = [] # type: ignore 

191 html = AUTO_REF_RE.sub(fix_ref(url_mapper, unmapped), html) 

192 return html, unmapped
